//decimal to binary using function
var decimal=11;
function converter(decimal){
  if(decimal==0){
    return 0;
  }
  else if(decimal==1){
    return 1;
  }
  else{
    return converter(parseInt(decimal/2)) + (decimal%2).toString();
  }
}

//concatenating string

var a="Apple";

}

//Concadinating more than two strings

var str="Aaaaa Bbb Oooooooo Pppp"
var strArr=String(str).split(" ");
var max=strArr[0].length;
for(var i=1;i<strArr.length;i++){
  if(max<strArr[i].length){
    max=strArr[i].length;
  }
}
for(var j=0;j<max;j++){
  for(var k=0;k<strArr.length;k++){
    if(j<strArr[k].length){
      process.stdout.write(strArr[k][j]+" ");
    }
  }
}
